\section*{Main result}

\begin{proposition}
The orthogonal complement of a subspace is a subspace.
\end{proposition}

\begin{proposition}
Let $L \subset \R ^n$ be a subspace. Then
\[
\dim L + \dim L^{\perp } = n.
\]
\end{proposition}

\begin{proposition}
Let $b_1, \dots , b_m$ be a basis for a subspace $L \subset \R ^n$. Then $x \perp L$ if and only if $x \perp b_i$ for $i = \upto{m}$.
\end{proposition}
