  \name{$z$}
  \have{a}{$x \subset y$}
  \have{b}{$y \subset z$}
  \thus{c}{$(\forall a)(a \in x \implies a \in y)$}{\ref{a}}
  \thus{d}{$(\forall a)(a \in y \implies a \in z)$}{\ref{b}}

\end{account}
