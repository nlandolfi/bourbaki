\stitle{Cumulative Distribution Function From Density}

\ssection{Why}

The density of a random
variable determines
the cumulative distribution
function.

\ssection{Result}

\begin{prop}
A probability density
function of a random variable,
if it exists,
characterizes the cumulative
distribution function.

\begin{proof}
Let $(X, \SA, \mu)$ be a
probability space.
Let $f$ be a real-valued
random variable on $X$.
Let $\lambda$ denote
the cover length.
Let $g$ be a probability
density of $f$:
Then:
\[
\begin{aligned}
    \rvcdf{f}(t)
    &= \mu(\Set*{x \in X}{f(x) \in (-\infty, t])}) \\
    &= \int_{(-\infty, t]} gd\lambda.
\end{aligned}
\]
\end{proof}

\end{prop}
