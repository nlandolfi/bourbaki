\ssection{Why}

How does arithmetic interact with integers.

\ssection{Results}

\begin{proposition}
  Let $a, b, c, d \in \Z$.
  If $a \leq b$ and $c \leq d$, then $a + b \leq c + d$.
\end{proposition}

\begin{proposition}
  Let $a, b, c, d \in \Z$ with $a, b \geq 0_{\Z}$.
  If $a \leq b$ and $c \leq d$, then $a \cdot c \leq a \cdot d$.
\end{proposition}
