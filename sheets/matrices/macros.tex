% allow ddots at any angle
% \dddots{1}{1} produces dots at 45 degrees
% \dddots{2}{1} produces dots at 30 degrees
% and so on
\def\dddots#1#2{%
  \mbox{%
    \unitlength 0.083333in%r
    \begin{picture}(#1,#2)(0.25,0)
      \unitlength 0.0416666in%
      \multiput(0,2)(#1,-#2){3}{.}
    \end{picture}%
    }%
  }
