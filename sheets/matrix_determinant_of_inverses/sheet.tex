
\section*{Result}

\begin{proposition}
The determinant of an invertible square real matrix is the multiplicative inverse of the determinant of the matrix.

\begin{proof}Let $A \in \R ^{n \times  n}$ be invertible.
We want to show that
\[
\det (A^{-1}) = (det(A))^{-1}.
\]
\end{proof}
\end{proposition}
