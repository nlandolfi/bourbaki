\section*{Why}

Under which operations is the set of measurable functions closed?

\section*{Overview}

Measurable functions are closed under composition and concatenation.
Taking these facts together with the observation that continuous functions are measurable, we conclude that measurable functions are closed under addition, multiplication, and (with suitable nonzero assumptions) division.

\section*{Results}

\begin{proposition}
The composition of two
measurable functions
is measurable.
\end{proposition}
\begin{proof}Let
$(X, \mathcal{A} )$,
$(Y, \mathcal{B} )$,
$(Z, \mathcal{C} )$
be measurable spaces.
Let $f: X \to Y$
and $g: Y \to Z$ be
measurable functions.
Define $h = g \circ f$.

Let $C \in \mathcal{C} $.
Measurability of $g$
implies
$g^{-1}(C) \in \mathcal{B} $.
This fact, together with
measurability of $f$,
implies
$f^{-1}(g^{-1}(C)) \in \mathcal{A} $.
Since $h^{-1} = f^{-1} \circ g^{-1}$,
we conclude that
$h$ is measurable.
\end{proof}
\begin{proposition}
A continuous function
between topological
spaces
is measurable with
respect to the
topological sigma algebras.
\end{proposition}
\begin{proposition}
The concatentation of
two measurable functions
is measurable.
\end{proposition}
\begin{proof}Let
$(X, \mathcal{A} )$,
$(Y, \mathcal{B} )$
and
$(Z, \mathcal{C} )$
be measurable spaces.
Let
$f: X \to Y$
and
$g: X \to Z$
be measurable.
Define
$h: X \to Y \cross Z$
by $h(x) = (f(x), g(x))$.\end{proof}
\begin{proposition}
Let $(X, \mathcal{A} )$
be a measurable space
and let $R$ denote the
real numbers.
Let
$f,g: X \to R$
be measurable.

Then, $f+g$ and $fg$ are measurable.
\end{proposition}
