%!name:complete_inner_product_decomposition
%!need:orthogonal_complements

\begin{prop}
Let $(V, F)$ be a complete
inner product space.
Let $M$ be a closed subspace.
then
\[
  H = M \oplus M^{\perp}.
\]
That is for all $x \in V$,
there exists unique
$z \in M$ and $q \in M^{\perp}$
such that $x = z + w$.
\end{prop}
