\section*{Main result}

\begin{proposition}
$M \subset \R ^{n}$ is affine if and only if there exists $b \in \R ^m$ and $B \in \R ^{m \times n}$ such that
  \[
M = \Set*{x \in \R ^{n}}{Bx = b}.
  \]
\end{proposition}
