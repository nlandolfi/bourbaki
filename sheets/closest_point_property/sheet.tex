\section{Result}
\begin{proposition}
Suppose $H$ is a Hilbert space, $A \subset H$ closed and convex, and $x \in H$.
There exists a unique $z \in A$ satisfying
  \[
d(z, x) = \inf_{y \in A} d(y, x).
  \]
\end{proposition}
