\section{Result}
\begin{proposition}
Suppose $X$ is a complete inner product space, $A \subset X$ closed and convex, and $x \in X$.
There exists a unique $z \in X$ satisfying
  \[
d(z, x) = \inf_{y \in A} d(y, x).
  \]
\end{proposition}
