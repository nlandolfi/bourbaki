\ssection{Result}

\begin{prop}
The expectation of
the product of
two independent
random variables
is the product of
their expectations.
\begin{proof}
Let $f, g$ be two
independent random variables.
We want to show
\[
  \Expect fg = \Expect f \Expect g.
\]
\end{proof}
\end{prop}
