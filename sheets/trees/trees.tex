\sstart
\stitle{Trees}

\ssection{Why}

Tree branches split
and do not recombine.
We formalize this
property in the
language of graphs.

\ssection{Definition}

A
\ct{tree}{tree}
is a
connected
acyclic graph.

\ssubsection{Notation}

We denote trees by $T$,
a mnemonic for \say{tree.}
Let $u, v$ be two vertices
connected in $T$.
We denote that the edge
between $u$ and $v$ exists
by writing $\set{u, v} \in T$.

\ssection{Properties}

\begin{prop}
  There is only one
  path between
  any two vertices in a tree.

\begin{proof}
  Suppose to the contrary that
  there were two paths from
  vertex $u$ to vertex $v$,
  then by combining these paths
  we obtain a cycle. But the
  tree has no cycles,
  so there must not be two paths
  between any two vertices.
\end{proof}
\end{prop}
