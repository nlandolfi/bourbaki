  \slidei{ Norm }{
  \item for $x \in \R^n$ we define the (Euclidean) norm as
    \[
\norm{x}
= \sqrt{x_1^2 + x_2^2 + \cdots + x_n^2}
= \sqrt{x^\tp x}
    \]
  \item $\norm{x}$ measures length of vector from origin
  \item important properties
    \begin{itemize}
  \item $\norm{\alpha x} = \abs{\alpha }\norm{x}$
  \item $\norm{x+y} = \norm{x} + \norm{y}$
  \item $\norm{x} \geq 0$
  \item $\norm{x} = 0 \iff x = 0$
    \end{itemize}
}
