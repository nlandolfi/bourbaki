\ssubsection{Defining Result}

\begin{prop}
Let $(X, \SA)$
be a measurable space.
Let $\mu,\nu$
be sigma-finite measures
with $\nu \ll \mu$.
Then there exists
$g: X \to [0, \infty)$
such that
\[
  \nu(A) = \int_A g d\mu
\]
for all $A \in \SA$.
